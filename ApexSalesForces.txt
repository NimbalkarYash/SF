public class StudentManager {


    // ===== Controller Variables =====
    public String sid {get; set;}
    public String name {get; set;}
    public String cls {get; set;}
    public Decimal gpa {get; set;}


    public Student__c studentRecord {get; set;}
    public List<Student__c> allStudents {get; set;}


    // ===== Insert Student =====
    public static void addStudent(String sid, String name, String cls, Decimal gpa) {


        Student__c s = new Student__c();
        s.Student_ID__c = sid;
        s.Name = name;
        s.Class__c = cls;
        s.GPA__c = gpa;


        insert s;
    }


    // ===== Get student by ID =====
    public static Student__c getStudentById(String sid) {


        List<Student__c> students = [
            SELECT Id, Name, Student_ID__c, Class__c, GPA__c
            FROM Student__c
            WHERE Student_ID__c = :sid
        ];


        if(students.size() > 0) {
            return students[0];
        }


        return null;
    }


    // ===== Delete student =====
    public static void deleteStudent(String sid) {


        List<Student__c> students = [
            SELECT Id
            FROM Student__c
            WHERE Student_ID__c = :sid
        ];


        if(students.size() > 0) {
            delete students[0];
        }
    }


    // ===== Update Student =====
    public static void updateStudent(String sid, String name1, String class1, Decimal gpa1) {


        List<Student__c> students = [
            SELECT Id, Name, Student_ID__c, Class__c, GPA__c
            FROM Student__c
            WHERE Student_ID__c = :sid
        ];


        if(students.size() > 0) {


            Student__c s = students[0];
            s.Name = name1;
            s.Class__c = class1;
            s.GPA__c = gpa1;


            update s;
        }
    }


    // ===== Get All Students =====
    public static List<Student__c> getAllStudents() {


        return [
            SELECT Id, Name, Student_ID__c, Class__c, GPA__c
            FROM Student__c
        ];
    }


    // ===== Controller Methods =====


    public void addStudentVF() {
        addStudent(sid, name, cls, gpa);
    }


    public void getStudentVF() {
        studentRecord = getStudentById(sid);
    }


    public void deleteStudentVF() {
        deleteStudent(sid);
        studentRecord = null;
    }


    public void updateStudentVF() {
        updateStudent(sid, name, cls, gpa);
    }


    public void getAllStudentsVF() {
        allStudents = getAllStudents();
    }


}
