public class StudentController {


    // Input Variables
    public String studentName {get; set;}
    public Integer rollNo {get; set;}
    public String email {get; set;}
    public Decimal marks {get; set;}


    // Message
    public String msg {get; set;}


    // Student List
    public List<Student__c> studentList {get; set;}


    // Save Student
    public void saveStudent() {


        try {


            Student__c s = new Student__c();


            s.Name = studentName;
            s.Roll_Number__c = rollNo;
            s.Email__c = email;
            s.Marks__c = marks;


            insert s;


            msg = 'Student Record Saved Successfully';


            loadStudents();


            // Clear fields
            studentName = '';
            rollNo = null;
            email = '';
            marks = null;


        }


        catch(Exception e) {


            msg = e.getMessage();
        }
    }


    // Load All Students
    public void loadStudents() {


        studentList = [
            SELECT Id,
                   Name,
                   Roll_Number__c,
                   Email__c,
                   Marks__c
            FROM Student__c
        ];
    }


    // Constructor
    public StudentController() {


        loadStudents();
    }
}
