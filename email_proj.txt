public class email_proj {
    public String email {get; set;}
    public String msg {get; set;}
    public Blob filebody {get; set;}
    public string filename {get; set;}
   
    public void sendEmail() {


        Messaging.SingleEmailMessage mail = new Messaging.SingleEmailMessage();


        if(String.isBlank(email) || !email.contains('@')){
            msg = 'Invalid Email ID';
            return;
        }
        mail.setToAddresses(new String [] {email});
        mail.setSubject('Test Email and file transfer from Salesforce Apex');
        mail.setPlainTextBody('Hello, this is a test email with file sent using Apex.');
       
        if(String.isBlank(filename) || filebody == null){
            msg = 'File not Uploaded Properly';
            return;
        }
       
        Messaging.EmailFileAttachment attach = new Messaging.EmailFileAttachment();
        attach.setfileName(filename);
        attach.setBody(filebody);
       
        mail.setFileAttachments(new Messaging.EmailFileAttachment[] {attach});


        Messaging.SendEmailResult[] results =
            Messaging.sendEmail(new Messaging.SingleEmailMessage[] {mail});


        if(results[0].isSuccess()){
            msg = 'Email Sent Successfully';
        }
        else{
            msg = 'Email Failed: ' + results[0].getErrors()[0].getMessage();
        }
    }
}
















<apex:page controller="email_proj">
    <apex:form enctype="multipart/form-data">
        <apex:pageBlock title="Send Email Application">
            <apex:pageBlockSection columns="1">
                <apex:inputText label="Enter Email Address" value="{!email}" required="true"/>
                <apex:inputFile value="{!filebody}" filename="{!filename}"/>
                <apex:commandButton value="Send Email" action="{!sendEmail}"/>
            </apex:pageBlockSection>
            <apex:outputPanel id="msgPanel">
                <apex:pageMessage summary="{!msg}" severity="info" rendered="{!NOT(ISBLANK(msg))}"/>
            </apex:outputPanel>
        </apex:pageBlock>
    </apex:form>
</apex:page>
